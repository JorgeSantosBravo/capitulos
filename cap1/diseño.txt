CREATE TABLE `capitulos`.`capitulo` ( `id_capitulo` INT(4) NOT NULL , `fecha` DATE NOT NULL , `medio` VARCHAR(15) NOT NULL , `formato` VARCHAR(15) NOT NULL , `serie` INT(3) NOT NULL , `s` INT(3) NOT NULL , `e` INT(4) NOT NULL , `Titulo` VARCHAR(50) NOT NULL , `Director` INT(4) NOT NULL , `Duracion` INT(3) NOT NULL , `Comentario` TEXT NOT NULL , PRIMARY KEY (`id_capitulo`)) ENGINE = InnoDB;
